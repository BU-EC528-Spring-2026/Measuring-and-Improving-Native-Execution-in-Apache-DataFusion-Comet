// STEP 0:
cd ~/datafusion-comet

export SPARK_HOME=~/spark-3.5.8-bin-hadoop3
export JAVA_HOME=$(/usr/libexec/java_home -v 17)
export COMET_JAR=$(ls spark/target/comet-spark-spark3.5_2.12-*.jar | grep -v sources | head -n 1)

$SPARK_HOME/bin/spark-shell \
  --jars "$COMET_JAR" \
  --conf spark.driver.extraClassPath="$COMET_JAR" \
  --conf spark.executor.extraClassPath="$COMET_JAR" \
  --conf spark.plugins=org.apache.spark.CometPlugin \
  --conf spark.comet.enabled=true \
  --conf spark.comet.explainFallback.enabled=true \
  --conf spark.comet.caseConversion.enabled=true \
  --conf spark.comet.operator.DataWritingCommandExec.allowIncompatible=true \
  --conf spark.comet.operator.WindowExec.allowIncompatible=true \
  --conf spark.comet.exec.sortMergeJoinWithJoinFilter.enabled=true \
  --conf spark.memory.offHeap.enabled=true \
  --conf spark.memory.offHeap.size=16g \
  --conf spark.comet.parquet.write.enabled=true \
  --conf spark.comet.exec.localTableScan.enabled=true \
  --conf spark.shuffle.manager=org.apache.spark.sql.comet.execution.shuffle.CometShuffleManager

// STEP 1: Create Parquet data first (write once)
spark.sql("""
  SELECT 
    id,
    cast(id % 3 as string) as category,
    cast(id % 2 as string) as partition_col,
    rand() * 100 as value,
    id as order_col
  FROM range(1000)
""").write.mode("overwrite").parquet("/tmp/window_test_data")

// STEP 2: Read from Parquet for all tests
val windowDF = spark.read.parquet("/tmp/window_test_data")
windowDF.createOrReplaceTempView("window_tbl")

println("=== VERIFY SOURCE ===")
spark.sql("SELECT * FROM window_tbl LIMIT 5").show()
spark.sql("EXPLAIN SELECT * FROM window_tbl").show()

// STEP 3: Now test ALL window functions (no Range in sight)
println("\n--- RANKING FUNCTIONS ---")
spark.sql("SELECT category, value, ROW_NUMBER() OVER (PARTITION BY category ORDER BY value) FROM window_tbl").explain()
spark.sql("SELECT category, value, RANK() OVER (PARTITION BY category ORDER BY value) FROM window_tbl").explain()
spark.sql("SELECT category, value, DENSE_RANK() OVER (PARTITION BY category ORDER BY value) FROM window_tbl").explain()
spark.sql("SELECT category, value, PERCENT_RANK() OVER (PARTITION BY category ORDER BY value) FROM window_tbl").explain()
spark.sql("SELECT category, value, NTILE(4) OVER (PARTITION BY category ORDER BY value) FROM window_tbl").explain()
spark.sql("SELECT category, value, CUME_DIST() OVER (PARTITION BY category ORDER BY value) FROM window_tbl").explain()

println("\n--- VALUE FUNCTIONS ---")
spark.sql("SELECT category, value, LAG(value) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, LAG(value, 2) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, LAG(value, 1, 0) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, LEAD(value) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, LEAD(value, 2) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, LEAD(value, 1, 0) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()

println("\n--- FIRST/LAST VALUE ---")
spark.sql("SELECT category, value, FIRST_VALUE(value) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, LAST_VALUE(value) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, FIRST_VALUE(value, true) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, NTH_VALUE(value, 2) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()

println("\n--- AGGREGATE WINDOW FUNCTIONS ---")
spark.sql("SELECT category, value, SUM(value) OVER (PARTITION BY category) FROM window_tbl").explain()
spark.sql("SELECT category, value, AVG(value) OVER (PARTITION BY category) FROM window_tbl").explain()
spark.sql("SELECT category, value, MIN(value) OVER (PARTITION BY category) FROM window_tbl").explain()
spark.sql("SELECT category, value, MAX(value) OVER (PARTITION BY category) FROM window_tbl").explain()
spark.sql("SELECT category, value, COUNT(value) OVER (PARTITION BY category) FROM window_tbl").explain()
spark.sql("SELECT category, value, STDDEV(value) OVER (PARTITION BY category) FROM window_tbl").explain()
spark.sql("SELECT category, value, VARIANCE(value) OVER (PARTITION BY category) FROM window_tbl").explain()

println("\n--- ORDERED WINDOWS ---")
spark.sql("SELECT category, value, SUM(value) OVER (PARTITION BY category ORDER BY id) FROM window_tbl").explain()
spark.sql("SELECT category, value, AVG(value) OVER (PARTITION BY category ORDER BY id ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING) FROM window_tbl").explain()
spark.sql("SELECT category, value, SUM(value) OVER (PARTITION BY category ORDER BY id ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) FROM window_tbl").explain()
spark.sql("SELECT category, value, SUM(value) OVER (PARTITION BY category ORDER BY id ROWS BETWEEN CURRENT ROW AND UNBOUNDED FOLLOWING) FROM window_tbl").explain()

println("\n--- RANGE WINDOWS ---")
spark.sql("SELECT category, value, SUM(value) OVER (PARTITION BY category ORDER BY value RANGE BETWEEN 5 PRECEDING AND CURRENT ROW) FROM window_tbl").explain()
